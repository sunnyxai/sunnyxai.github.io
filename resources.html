Binary Search

Classic O(log n) search on a sorted array.

def binary_search(arr, target):
    """Return index of target in sorted arr, or -1 if not found."""
    low, high = 0, len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
    return -1

# Usage
numbers = [1, 3, 5, 7, 9, 11, 13]
print(binary_search(numbers, 7))  # Output: 3

Simple Linear Regression

Train a model and visualise predictions with scikit-learn.

import numpy as np
from sklearn.linear_model import LinearRegression
import matplotlib.pyplot as plt

# Generate sample data
X = np.array([[1], [2], [3], [4], [5]])
y = np.array([2.1, 4.0, 5.8, 8.1, 10.2])

# Train the model
model = LinearRegression()
model.fit(X, y)

# Predict
predictions = model.predict(X)
print(f"Slope: {model.coef_[0]:.2f}, Intercept: {model.intercept_:.2f}")

# Plot
plt.scatter(X, y, color="#00f3ff", label="Data")
plt.plot(X, predictions, color="#bc13fe", label="Fit")
plt.legend()
plt.show()

Fibonacci — Memoization

Top-down dynamic programming approach using functools cache.

from functools import lru_cache

@lru_cache(maxsize=None)
def fibonacci(n):
    """Return the n-th Fibonacci number (0-indexed)."""
    if n < 2:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

# Print first 10 Fibonacci numbers
for i in range(10):
    print(f"F({i}) = {fibonacci(i)}")
